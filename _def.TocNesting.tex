%\newlinechar=`\^^J
%\def\puts#1{\message{#1 ^^J}}

\newcount\c
\def\tocNesting#1{%
	  \def\endUL{\HCode{</ul>}}
      \expandafter\ifx \csname level#1\endcsname\relax
          %\ifnum #1>0 \HCode{<ul class="collapsibleList">}\fi
          %\ifnum #1=0 \HCode{<ul class='treeView'>}\fi
          \ifnum #1>0 \HCode{<ul class='collapsibleList'>}\fi \\
          %\ifnum #1=0 \HCode{<ul class='collapsibleList'>}\fi \\
          \expandafter\def \csname level#1\endcsname{\endUL}
      \fi
      \c=#1 \advance\c by 1
      \loop
		% Invoke \level$c
        \csname level\the\c\endcsname

		% Empty the \level$c command
        \expandafter\let \csname level\the\c\endcsname\relax

		% $c++
        \advance\c by 1
      \ifnum \c<10   \repeat
}
